创建一个simpletab类，该类继承sdltool.tab_sample的init和bindChange方法。实例化一个simpletab对象，传递一个id参数（html结构中ul的id值），并自动调用init方法实现切换效果

            var simpletab=sdltool.tab_sample.extend({
                init:function(id) {
                    var that=this;
                    that._super(id);
                    that.bindChange();
                },
                bindChange:function(){
                    var that=this;
                    that._super();
                }
            }); 
            var  tab=new simpletab('datetype');
